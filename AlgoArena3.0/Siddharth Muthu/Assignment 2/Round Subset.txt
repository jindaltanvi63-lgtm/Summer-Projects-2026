#include <bits/stdc++.h>
#define ll int
#define endl '\n'
#define rep(i, a, b) for(ll i = a; i < b; i++)
#define repb(i, a, b) for(ll i = a; i > b; i--)

using namespace std;


void print(vector<ll> a)
{
    for(auto i : a)
        cout<<i<<" ";
    cout<<endl;
}

void input(vector<ll> &a, ll n)
{
    for(int i = 0; i<n; i++)
    {
        cin>>a[i];
    }
}

void solve()
{
    ll n, k;
    cin>>n>>k;
    vector<ll> a(n), a2(n), a5(n);
    for(int i = 0; i<n; i++)
        cin>>a[i];
    for(int i = 0; i<n; i++)
    {
        while(a[i]%2 == 0)
        {
            a[i]/=2;
            a2[i]++;
        }
        while(a[i]%5 == 0)
        {
            a[i]/=5;
            a5[i]++;
        }
    }
    vector<vector<vector<ll>>> dp(n, vector<vector<ll>>(k+1, vector<ll> (6000, -100)));
    dp[0][1][a5[0]] = a2[0];
    for(int i = 0; i<n; i++)
    {
        for(int j = 1; j<k; j++)
        {
            if(i+1>=j)
                dp[i][j][0] = 0;
        }
    }
    for(int i = 1; i<n; i++)
    {
        for(int j = 1; j<=k; j++)
        {
            for(int l = 0; l<6000; l++)
            {
                if(l>=a5[i])
                    dp[i][j][l] = max({dp[i][j][l], dp[i-1][j][l], dp[i-1][j-1][l-a5[i]]+a2[i]});
                else
                    dp[i][j][l] = max(dp[i-1][j][l], dp[i][j][l]);
                //cout<<i<<" "<<j<<" "<<l<<" "<<dp[i][j][l]<<" "<<dp[i-1][j-1][1]<<endl;
            }
        }
    }
    ll ans = 0;
    for(int i = 0; i<6000; i++)
    {
        ans = max(ans, min((ll)i, dp[n-1][k][i]));
    }
    cout<<ans<<endl;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    int t = 1;
    //cin >> t;
    while (t--) solve();
    
    return 0;
}
