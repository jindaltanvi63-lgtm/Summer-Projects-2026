#include <bits/stdc++.h>
#define ll long long
#define endl '\n'
#define rep(i, a, b) for(ll i = a; i < b; i++)
#define repb(i, a, b) for(ll i = a; i > b; i--)

using namespace std;


void print(vector<ll> a)
{
    for(auto i : a)
        cout<<i<<" ";
    cout<<endl;
}

void input(vector<ll> &a, ll n)
{
    for(int i = 0; i<n; i++)
    {
        cin>>a[i];
    }
}

void solve()
{
    ll n1, n2, k1, k2;
    cin>>n1>>n2>>k1>>k2;
    //ll dp[n1+1][n2+1][max(k1, k2)][3] = {0};
    k1++;
    k2++;
    vector<vector<vector<vector<ll>>>> dp(n1+1, vector<vector<vector<ll>>> (n2+1, vector<vector<ll>>(max(k1, k2), vector<ll>(3))));
    ll MOD = 1e8;
    
    dp[1][0][1][1] = 1;
    dp[0][1][1][2] = 1;
    for(int i = 0; i<=n1; i++)
    {
        for(int j = 0; j<=n2; j++)
        {
            for(int k = 1; k<k1; k++)
            {
                if(i<n1 && k+1<k1)
                {
                    dp[i+1][j][k+1][1]+=dp[i][j][k][1];
                    dp[i+1][j][k+1][1]%=MOD;
                }
                if(j<n2)
                {
                    dp[i][j+1][1][2]+=dp[i][j][k][1];
                    dp[i][j+1][1][2]%=MOD;
                }
                //cout<<i<<" "<<j<<" "<<k<<" "<<dp[i][j][k][1]<<endl;
            }
            
            for(int k = 1; k<k2; k++)
            {
                if(j<n2 && k+1<k2)
                {
                    dp[i][j+1][k+1][2]+=dp[i][j][k][2];
                    dp[i][j+1][k+1][2]%+MOD;
                }
                if(i<n1)
                {
                    dp[i+1][j][1][1] += dp[i][j][k][2];
                    dp[i+1][j][1][1]%=MOD;
                }
                //cout<<i<<" "<<j<<" "<<k<<" "<<dp[i][j][k][2]<<endl;
            }
        }
    }
    ll ans = 0;
    for(int i = 1; i<k1; i++)
    {
        ans+=dp[n1][n2][i][1];
        //cout<<dp[n1][n2][i][1]<<" ";
    }
    //cout<<endl;
    for(int i = 1; i<k2; i++)
    {
        ans+=dp[n1][n2][i][2];
        //cout<<dp[n1][n2][i][2]<<" ";
    }
    cout<<ans%(ll)(1E8)<<endl;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    int t = 1;
    //cin >> t;
    while (t--) solve();
    
    return 0;
}
