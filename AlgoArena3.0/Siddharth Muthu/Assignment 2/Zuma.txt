#include <bits/stdc++.h>
#define ll long long
#define endl '\n'
#define rep(i, a, b) for(ll i = a; i < b; i++)
#define repb(i, a, b) for(ll i = a; i > b; i--)

using namespace std;


void print(vector<ll> a)
{
    for(auto i : a)
        cout<<i<<" ";
    cout<<endl;
}

void input(vector<ll> &a, ll n)
{
    for(int i = 0; i<n; i++)
    {
        cin>>a[i];
    }
}

void solve()
{
    ll n;
    cin>>n;
    vector<ll>a(n);
    for(int i = 0; i<n; i++)
        cin>>a[i];
    vector<vector<ll>> dp(n, vector<ll> (n, 1000));
    for(int i = 0; i<n; i++)
    {
        dp[i][i] = 1;
    }
    for(int j = 0; j<n; j++)
    {
        for(int i = j-1; i>=0; i--)
        {
            for(int k = i; k<j; k++)
            {
                dp[i][j] = min(dp[i][j], dp[i][k]+dp[k+1][j]);
            }
            if(a[i] == a[j])
            {
                if(i+1 == j)
                    dp[i][j] = 1;
                else
                    dp[i][j] = min(dp[i][j], dp[i+1][j-1]);
            }
        }
    }
    // for(int i = 0; i<n; i++)
    //     print(dp[i]);
    cout<<dp[0][n-1]<<endl;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    int t = 1;
    //cin >> t;
    while (t--) solve();
    
    return 0;
}
