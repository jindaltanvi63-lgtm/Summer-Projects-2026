const ll MOD = 998244353;

ll C(ll n, ll r)
{
    ll ans = 1;
    for(int i = n; i>n-r; i--)
    {
        ans*=i;
        ans/=n-i+1;
    }
    return ans;
}

vector<ll> divisor(1e6+1, -1);
void sieve()
{
    for(int i = 2; i<divisor.size(); i++)
    {
        if(divisor[i] == -1){
        for(int j = i*2; j<divisor.size(); j+=i)
        {
            divisor[j] = i;
        }
    }
    }
}


ll mkpow(ll a, ll b)
{
    if(b == 0)
        return 1;
    ll t = (mkpow(a, b/2));
    //cout<<a<<" "<<b<<" "<<t<<endl;
    if(b%2 == 0)
        return (t*t)%MOD;
    return (((t*t)%MOD)*a)%MOD;
}
vector<bool> vis;
set<ll> store;
ll sz;
map<ll, vector<pair<ll, ll>>> adj;
void dfs(ll node, ll type)
{
    vis[node] = true;
    sz++;
    store.insert(node);
    for(auto i: adj[node])
    {
        if(i.second == type && !vis[i.first] )
            dfs(i.first, type);
    }
}



void solve() 
{
    ll n, m, sz = 0;
    cin>>n>>m;
    vector<string>a(n);
    vector<ll> zeroes;
    for(int i = 0; i<n; i++)
    {
        cin>>a[i];
        ll j = a[i].size()-1, z = 0;
        while(a[i][j] == '0')
        {
            z++;
            j--;
        }
        zeroes.push_back(z);
        sz+=a[i].size();
    }
    sort(zeroes.begin(), zeroes.end(), greater<char>());
    ll sum = 0;
    for(int i = 0; i<zeroes.size(); i+=2)
        sum+=zeroes[i];
    if(sz-sum >= m+1)
    {
        cout<<"Sasha"<<endl;
    }
    else
        cout<<"Anna"<<endl;
}
