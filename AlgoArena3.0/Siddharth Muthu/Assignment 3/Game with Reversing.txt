const ll MOD = 998244353;

ll C(ll n, ll r)
{
    ll ans = 1;
    for(int i = n; i>n-r; i--)
    {
        ans*=i;
        ans/=n-i+1;
    }
    return ans;
}

vector<ll> divisor(1e6+1, -1);
void sieve()
{
    for(int i = 2; i<divisor.size(); i++)
    {
        if(divisor[i] == -1){
        for(int j = i*2; j<divisor.size(); j+=i)
        {
            divisor[j] = i;
        }
    }
    }
}


ll mkpow(ll a, ll b)
{
    if(b == 0)
        return 1;
    ll t = (mkpow(a, b/2));
    //cout<<a<<" "<<b<<" "<<t<<endl;
    if(b%2 == 0)
        return (t*t)%MOD;
    return (((t*t)%MOD)*a)%MOD;
}
vector<bool> vis;
set<ll> store;
ll sz;
map<ll, vector<pair<ll, ll>>> adj;
void dfs(ll node, ll type)
{
    vis[node] = true;
    sz++;
    store.insert(node);
    for(auto i: adj[node])
    {
        if(i.second == type && !vis[i.first] )
            dfs(i.first, type);
    }
}

ll ans(string s1, string s2)
{
    ll final = 1e18;
    ll count1 = 0, count2 = 0;
    for(int i = 0; i<s1.size(); i++)
    {
        if(s1[i]!=s2[i])
            count1++;
    }
    string s3 = s1;
    reverse(s1.begin(), s1.end());
    for(int i = 0; i<s1.size(); i++)
    {
        if(s1[i]!=s2[i])
            count2++;
    }
    if(count1 == 0)
        final = min(final, 0LL);
    else if(count1%2 == 0)
    {
        if(s1 == s3)
            final = min(final, 2*count1-1);
        else
            final = min(final, 2*count1);
    }
    else
        final = min(final, 2*count1-1);
    if(count2 == 0)
        final = min(final, 2LL);
    else if(count2%2 == 0)
    {
        final = min(final, count2*2-1);
    }
    else
    {
        if(s1 == s3)
            final = min(final, count2*2-1);
        else
            final = min(final, 2*count2);
    }
    return final;

}

void solve() 
{
    ll n;

    string s1, s2;
    cin>>s1>>s2;
    cout<<min(ans(s1, s2), ans(s2, s1))<<endl;
}
