bool prime(ll n)
{
    for(int i = 2; i<=sqrt(n); i++)
    {
        if(n%i == 0)
            return false;
    }
    return true;
}

void solve()
{
    ll n;
    cin>>n;
    vector<ll>a(n);
    for(int i = 0; i<n; i++)
    {
        cin>>a[i];
    }
    vector<ll>b = a;
    sort(b.begin(), b.end());
    vector<ll>dpb(n+1), dpa(n+1);
    for(int i = 1; i<=n; i++)
    {
        dpa[i] = dpa[i-1]+a[i-1];
        dpb[i] = dpb[i-1]+b[i-1];
    }
    ll m;
    cin>>m;
    for(int i = 0; i<m; i++)
    {
        ll type, l, r;
        cin>>type>>l>>r;
        if(type == 1)
        {
            cout<<dpa[r] - dpa[l-1]<<endl;
        }
        else
            cout<<dpb[r] - dpb[l-1]<<endl;
    }
}
