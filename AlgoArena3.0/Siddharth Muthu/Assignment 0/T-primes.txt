bool prime(ll n)
{
    for(int i = 2; i<=sqrt(n); i++)
    {
        if(n%i == 0)
            return false;
    }
    return true;
}

void solve()
{
    ll n;

    if(floor(sqrt(n))*sqrt(n) == n && n!=1 && prime(floor(sqrt(n))))

        cout<<"NO"<<endl;
}
