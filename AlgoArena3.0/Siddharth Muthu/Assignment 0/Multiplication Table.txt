bool prime(ll n)
{
    for(int i = 2; i<=sqrt(n); i++)
    {
        if(n%i == 0)
            return false;
    }
    return true;
}

ll n, m, k;

ll ans(ll c)
{
    ll final = 0;
    for(int i = 1; i<=n; i++)
    {
        final+=min(c/i, m);
    }
    return final;
}

void solve()
{
    cin>>n>>m>>k;
    ll lo = 1, hi = 1e12, mid = (lo+hi)/2;
    while(hi>lo)
    {
        if(ans(mid)>=k)
        {
            hi = mid;
            mid = (lo+hi)/2;
        }
        else
        {
            lo = mid+1;
            mid = (hi+lo)/2;
        }
    }
    cout<<hi<<endl;
}
